# v0.5 — Conflict Resolver Implementation Plan (Sprint 2 of 2)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`) syntax for tracking.
>
> **Git policy:** subagents do NOT run git operations (`git add`, `git commit`, `git push`). Each task leaves changes unstaged; the maintainer decides on commits.

**Goal:** Заменить текущее поведение Push при merge-конфликте (`git rebase --abort` + сообщение «Resolve manually in...») на интерактивный резолвер: read-only three-way diff (base / remote / mine), per-file выбор «Mine / Remote / Open in editor», recovery-banner на cold start, completion через `git rebase --continue` → `git push`.

**Architecture:**
Main process получает новый модуль `src/main/conflict.ts` с функциями инспекции paused-rebase состояния, per-file разрешения (через `git checkout --ours/--theirs`), и `continue`/`abort`. `push.ts` теперь НЕ делает авто-`rebase --abort` — возвращает `RunResult` с `kind: 'conflict'`. Renderer открывает `ConflictModal` (новый), который рендерит `ConflictFileList` (левая панель) + `ThreeWayDiff` (центр, три колонки `<pre>`). На cold start `useAppState` проверяет состояние paused rebase и показывает persistent-баннер для recovery.

**Tech Stack:** TypeScript, Electron, React 18. `git` CLI. Тесты: Vitest + реальный `git init` в tmp-папке (детерминированные конфликты).

**Reference spec:** [docs/superpowers/specs/2026-05-08-v0.5-i18n-and-conflicts.md](../specs/2026-05-08-v0.5-i18n-and-conflicts.md)

## Терминологическая ловушка `git rebase`

Внутри `git pull --rebase` git инвертирует `--ours`/`--theirs`:

| User mental model     | git rebase term | git index stage | Команда                       |
|-----------------------|-----------------|-----------------|-------------------------------|
| **Mine** (это машина) | `--theirs`      | stage `:3:`     | `git checkout --theirs <p>`   |
| **Remote** (GitHub)   | `--ours`        | stage `:2:`     | `git checkout --ours <p>`     |
| **Base** (общий пред) | —               | stage `:1:`     | `git show :1:<p>`             |

Все константы и вызовы git проходят через явные имена `STAGE_BASE = 1`, `STAGE_REMOTE = 2`, `STAGE_MINE = 3` — чтобы в коде нельзя было перепутать.

```ts
import { existsSync, readFileSync } from 'node:fs'
import { join } from 'node:path'
import { execSync, spawnSync } from 'node:child_process'
import type {
  ConflictState,
  ConflictFile,
  ConflictFileStatus,
} from '@shared/api'

export const STAGE_BASE = 1
export const STAGE_REMOTE = 2  // git's --ours during rebase (= upstream)
export const STAGE_MINE = 3    // git's --theirs during rebase (= local commits)

function isRebaseInProgress(repoPath: string): boolean {
  return (
    existsSync(join(repoPath, '.git', 'rebase-merge')) ||
    existsSync(join(repoPath, '.git', 'rebase-apply'))
  )
}

function gitOutput(args: string[], cwd: string): string {
  const r = spawnSync('git', args, { cwd, encoding: 'utf8' })
  if (r.status !== 0) throw new Error(`git ${args.join(' ')} failed: ${r.stderr}`)
  return r.stdout
}

function gitTry(args: string[], cwd: string): { ok: boolean; stdout: string; stderr: string } {
  const r = spawnSync('git', args, { cwd, encoding: 'utf8' })
  return { ok: r.status === 0, stdout: r.stdout ?? '', stderr: r.stderr ?? '' }
}

function listUnmergedFiles(repoPath: string): string[] {
  // `git diff --name-only --diff-filter=U` lists paths with unmerged entries
  const out = gitOutput(['diff', '--name-only', '--diff-filter=U', '-z'], repoPath)
  if (!out) return []
  // -z gives null-delimited output (handles paths with newlines/spaces)
  return out.split('\0').filter(Boolean)
}

export function isFileBinary(repoPath: string, path: string): boolean {
  // Heuristic: read first 8 KB from working tree, look for null byte.
  const fullPath = join(repoPath, path)
  if (!existsSync(fullPath)) {
    // Working tree may not have the file (delete-vs-modify conflict).
    // Fall back to git's own check via numstat against an empty tree.
    const r = gitTry(['diff', '--numstat', '--cached', '--', path], repoPath)
    // numstat shows '-' for binary files
    if (r.stdout.startsWith('-\t-\t')) return true
    return false
  }
  try {
    const fd = readFileSync(fullPath)
    const sample = fd.subarray(0, 8192)
    for (let i = 0; i < sample.length; i++) {
      if (sample[i] === 0) return true
    }
    return false
  } catch {
    return false
  }
}

function fileStatus(repoPath: string, path: string): ConflictFileStatus {
  // After resolution we expect the file to no longer be in `--diff-filter=U`.
  // For un-resolved (still listed by listUnmergedFiles), status is 'unresolved'.
  // After 'git add' (post checkout --ours/--theirs/manual), the file leaves U list.
  // We can't distinguish mine vs remote vs manual from git state alone — those
  // labels are recorded by the caller via setFileStatus (see Task 4). So this
  // function only returns 'unresolved' for files still in the U list.
  return 'unresolved'
}

export function getConflictState(repoPath: string): ConflictState {
  if (!isRebaseInProgress(repoPath)) {
    return { inProgress: false, files: [] }
  }
  const paths = listUnmergedFiles(repoPath)
  const files: ConflictFile[] = paths.map((p) => ({
    path: p,
    status: fileStatus(repoPath, p),
    binary: isFileBinary(repoPath, p),
  }))
  return { inProgress: true, files }
}
```

```ts
import type { ConflictFileContent } from '@shared/api'

export function getStageContent(
  repoPath: string,
  path: string,
  stage: number,
): ConflictFileContent {
  const binary = isFileBinary(repoPath, path)
  if (binary) return { text: null, binary: true }

  // git show :N:<path> prints content of stage N from the index.
  // Use spawnSync so we can capture stderr separately.
  const r = spawnSync('git', ['show', `:${stage}:${path}`], {
    cwd: repoPath,
    encoding: 'utf8',
  })
  if (r.status !== 0) {
    // Stage may not exist (e.g., add/add conflict has no :1: base) — return empty.
    return { text: '', binary: false }
  }
  return { text: r.stdout ?? '', binary: false }
}
```

```ts
import type { ConflictResolveChoice, ConflictResolveResult, LocalizedMessage } from '@shared/api'

const CONFLICT_MARKER_RE = /^(<{7}|={7}|>{7})/m

function fileHasConflictMarkers(repoPath: string, path: string): boolean {
  const fullPath = join(repoPath, path)
  if (!existsSync(fullPath)) return false
  try {
    const content = readFileSync(fullPath, 'utf8')
    return CONFLICT_MARKER_RE.test(content)
  } catch {
    return false
  }
}

export function resolveFile(
  repoPath: string,
  path: string,
  choice: ConflictResolveChoice,
): ConflictResolveResult {
  if (choice === 'mine' || choice === 'remote') {
    const flag = choice === 'mine' ? '--theirs' : '--ours'
    const co = gitTry(['checkout', flag, '--', path], repoPath)
    if (!co.ok) {
      return {
        ok: false,
        error: {
          key: 'conflict.error.checkoutFailed',
          params: { reason: co.stderr.trim() },
          fallback: co.stderr.trim(),
        },
      }
    }
  } else {
    // manual: file must already be edited and free of conflict markers
    if (fileHasConflictMarkers(repoPath, path)) {
      return { ok: false, error: { key: 'conflict.error.markersRemain' } }
    }
  }
  const add = gitTry(['add', '--', path], repoPath)
  if (!add.ok) {
    return {
      ok: false,
      error: {
        key: 'conflict.error.addFailed',
        params: { reason: add.stderr.trim() },
        fallback: add.stderr.trim(),
      },
    }
  }
  return { ok: true }
}
```

```ts
import type { RunResult } from '@shared/api'

export function continueRebase(repoPath: string): RunResult {
  // Use core.editor=true so git skips opening an editor for the commit message
  // during `rebase --continue` (would block in headless main process).
  const r = spawnSync(
    'git',
    ['-c', 'core.editor=true', 'rebase', '--continue'],
    { cwd: repoPath, encoding: 'utf8' },
  )
  if (r.status === 0) {
    return { ok: true, exitCode: 0 }
  }
  // Failed: either still has unresolved files, or new conflict surfaced
  // in subsequent commit. Caller can re-fetch state and re-render the modal.
  const stderr = (r.stderr ?? '').trim()
  return {
    ok: false,
    exitCode: r.status ?? -1,
    error: {
      key: 'conflict.error.continueFailed',
      params: { reason: stderr },
      fallback: stderr,
    },
  }
}

export function abortRebase(repoPath: string): void {
  // Best-effort. Errors silenced because "no rebase in progress" is expected
  // when called from recovery paths.
  spawnSync('git', ['rebase', '--abort'], { cwd: repoPath, encoding: 'utf8' })
}
```

```tsx
function lineSet(text: string | null): Set<string> {
  if (text == null) return new Set()
  return new Set(text.split('\n'))
}

function colorRow(line: string, baseLines: Set<string>, otherSide: Set<string>): string {
  if (baseLines.has(line)) return ''
  // Differs from base
  if (otherSide.has(line)) return 'bg-yellow-50 dark:bg-yellow-900/20'
  return 'bg-amber-100 dark:bg-amber-900/30'
}

export function ThreeWayDiff({ path, fileContent }: Props) {
  const t = useT()
  const { base, remote, mine } = fileContent

  // Binary or absent
  if (base.binary || remote.binary || mine.binary) {
    return (
      <div className="flex h-full items-center justify-center text-sm text-neutral-500">
        {t('conflict.diff.binary')}
      </div>
    )
  }

  const baseText = base.text ?? ''
  const remoteText = remote.text ?? ''
  const mineText = mine.text ?? ''
  const totalBytes = baseText.length + remoteText.length + mineText.length

  if (totalBytes > MAX_PREVIEW_BYTES) {
    return (
      <div className="flex h-full items-center justify-center text-sm text-neutral-500">
        {t('conflict.diff.tooLarge', { size: Math.round(totalBytes / 1024) })}
      </div>
    )
  }

  const baseSet = lineSet(baseText)
  const remoteLines = remoteText.split('\n')
  const mineLines = mineText.split('\n')
  const baseLinesArr = baseText.split('\n')
  const remoteSet = new Set(remoteLines)
  const mineSet = new Set(mineLines)

  // Synchronized scroll: bind scrollTop across the three columns.
  const colsRef = useRef<HTMLDivElement[]>([])
  const setRef = (i: number) => (el: HTMLDivElement | null) => {
    if (el) colsRef.current[i] = el
  }
  const onScroll = (sourceIdx: number) => (e: React.UIEvent<HTMLDivElement>) => {
    const top = e.currentTarget.scrollTop
    colsRef.current.forEach((el, i) => {
      if (i !== sourceIdx && el) el.scrollTop = top
    })
  }

  const renderColumn = (
    label: string,
    lines: string[],
    otherSide: Set<string>,
    idx: number,
  ) => (
    <div className="flex min-w-0 flex-1 flex-col border-r border-neutral-200 last:border-r-0 dark:border-neutral-700">
      <div className="border-b border-neutral-200 bg-neutral-100 px-2 py-1 text-xs font-semibold dark:border-neutral-700 dark:bg-neutral-800">
        {label}
      </div>
      <div
        ref={setRef(idx)}
        onScroll={onScroll(idx)}
        className="flex-1 overflow-auto font-mono text-xs"
      >
        <pre className="m-0 p-0">
          {lines.length === 0 || (lines.length === 1 && lines[0] === '') ? (
            <span className="block px-2 py-0.5 text-neutral-500">{t('conflict.diff.empty')}</span>
          ) : (
            lines.map((line, i) => (
              <span
                key={i}
                className={`block px-2 py-0.5 ${colorRow(line, baseSet, otherSide)}`}
              >
                {line || ' '}
              </span>
            ))
          )}
        </pre>
      </div>
    </div>
  )

  return (
```

```tsx
  remote: ConflictFileContent
  mine: ConflictFileContent
}

export function ConflictModal({ open, onClose, onContinued }: Props) {
  const t = useT()
  const [files, setFiles] = useState<ConflictFile[]>([])
  const [localStatus, setLocalStatus] = useState<Record<string, LocalStatus>>({})
  const [selectedPath, setSelectedPath] = useState<string | null>(null)
  const [content, setContent] = useState<FileContents | null>(null)
  const [busyPath, setBusyPath] = useState<string | null>(null)
  const [error, setError] = useState<LocalizedMessage | null>(null)
  const [continuing, setContinuing] = useState(false)

  const refresh = async () => {
    const state = await window.api.conflictGetState()
    if (!state.inProgress) {
      onContinued()
      return
    }
    // Merge git's authoritative status with our local "resolved-as-X" tracking.
    // git state only knows 'unresolved' (still in U list) or absent.
    const merged: ConflictFile[] = state.files.map((f) => ({
      ...f,
      status: localStatus[f.path] ?? f.status,
    }))
    setFiles(merged)
    if (selectedPath == null && merged.length > 0) {
      setSelectedPath(merged[0].path)
    }
  }

  useEffect(() => {
    if (open) {
      void refresh()
    } else {
      setFiles([])
      setLocalStatus({})
      setSelectedPath(null)
      setContent(null)
      setError(null)
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [open])

  // Load 3-way content when selection changes
  useEffect(() => {
    if (!selectedPath) {
      setContent(null)
      return
    }
    let cancelled = false
    void (async () => {
      const [base, remote, mine] = await Promise.all([
        window.api.conflictGetFile(selectedPath, 'base'),
        window.api.conflictGetFile(selectedPath, 'remote'),
        window.api.conflictGetFile(selectedPath, 'mine'),
      ])
      if (!cancelled) setContent({ base, remote, mine })
    })()
    return () => {
      cancelled = true
    }
  }, [selectedPath])

  const handleResolve = async (path: string, choice: ConflictResolveChoice) => {
    setBusyPath(path)
    setError(null)
    try {
      const r = await window.api.conflictResolveFile(path, choice)
      if (!r.ok) {
        setError(r.error)
        return
      }
      const newStatus: LocalStatus =
        choice === 'mine'
          ? 'resolved-mine'
          : choice === 'remote'
            ? 'resolved-remote'
            : 'resolved-manual'
      setLocalStatus((prev) => ({ ...prev, [path]: newStatus }))
      await refresh()
    } finally {
      setBusyPath(null)
    }
  }

        if (r.kind === 'conflict') {

          setLocalStatus({})
          await refresh()
          if (r.error) setError(r.error)
          return
        }
        if (r.error) setError(r.error)
        return
      }
      // Success — rebase done.
      onContinued()
    } finally {
      setContinuing(false)
    }
  }

    if (!window.confirm(t('conflict.modal.abortConfirm'))) return
    await window.api.conflictAbort()
    onClose()
  }

  if (!open) return null

  const allResolved = files.every((f) => f.status !== 'unresolved')
  const selectedFile = files.find((f) => f.path === selectedPath) ?? null

  return (
    <div className="fixed inset-0 z-20 flex items-center justify-center bg-black/40">
      <div className="flex h-[80vh] w-[1100px] flex-col rounded-lg bg-white shadow-lg dark:bg-neutral-900">
        <div className="flex items-center justify-between border-b border-neutral-200 px-4 py-3 dark:border-neutral-700">
          <div>
            <h2 className="text-base font-semibold">{t('conflict.modal.title')}</h2>
            <p className="text-xs text-neutral-500">{t('conflict.modal.description')}</p>
          </div>
          <div className="flex gap-2">
            <button
              onClick={() => void handleAbort()}
              className="rounded border border-neutral-300 px-3 py-1 text-sm hover:bg-neutral-100 dark:border-neutral-600 dark:hover:bg-neutral-700"
            >
              {t('conflict.modal.abort')}
            </button>
            <button
              onClick={() => void handleContinue()}
              disabled={!allResolved || continuing}
              className="rounded bg-blue-600 px-3 py-1 text-sm text-white hover:bg-blue-700 disabled:bg-neutral-400"
            >
              {t('conflict.modal.continue')}
            </button>
          </div>
        </div>
        {error && (
          <div className="border-b border-red-300 bg-red-50 px-4 py-2 text-sm text-red-900 dark:border-red-700 dark:bg-red-950 dark:text-red-200">
            {tMessage(t, error)}
          </div>
        )}
        <div className="flex flex-1 overflow-hidden">
          <ConflictFileList
            files={files}
            selectedPath={selectedPath}
            busyPath={busyPath}
            onSelect={setSelectedPath}
            onResolve={handleResolve}
            onOpenInEditor={handleOpenInEditor}
          />
          <div className="flex-1 overflow-hidden">
            {selectedFile && content ? (
              <ThreeWayDiff path={selectedFile.path} fileContent={content} />
            ) : (
              <div className="flex h-full items-center justify-center text-sm text-neutral-500">
                {/* Empty state — no file selected or content loading. */}
                {!selectedPath ? '' : '…'}
              </div>
            )}
          </div>
        </div>
      </div>
    </div>
  )
}
```
